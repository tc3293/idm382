            #include "thingProperties.h"

            // Define the pins for the ultrasonic sensor
            const int trigPin = 0; // Trig connected to Pin 0
            const int echoPin = 1; // Echo connected to Pin 1

            void setup() {
            Serial.begin(9600);
            delay(1500); 

            pinMode(trigPin, OUTPUT);
            pinMode(echoPin, INPUT);

            initProperties();
            ArduinoCloud.begin(ArduinoIoTPreferredConnection);
            }

            void loop() {
            ArduinoCloud.update();
  
            // Measure distance
            distance = measureDistance();
  
             // Print distance to Serial Monitor
            Serial.print("Distance: ");
            Serial.print(distance);
            Serial.println(" cm");

            // Check if trash is detected
            if (distance > 0 && distance < 20) { // Adjust threshold as needed
            trashAlert = "Trash Detect! Don't Forgot to empty it!";
             } else {
            trashAlert = "Empty :D";
             }
  
            delay(1000);
            }

            // Function to measure distance with ultrasonic sensor
            float measureDistance() {
            digitalWrite(trigPin, LOW);
            delayMicroseconds(2);

            digitalWrite(trigPin, HIGH);
            delayMicroseconds(10);
            digitalWrite(trigPin, LOW);

            long duration = pulseIn(echoPin, HIGH);

            float distanceCm = duration * 0.034 / 2;

            return distanceCm;
            }

What will it do?


Instead of opening its lid automatically, the garbage can will be revamped. The trash box will detect or monitor the system trash box remotely via an IOT cloud or app, displaying the bin height level and current condition (empty, half-full, full).

How will the end user control it?

The end user can monitor the trash level through an Arduino IoT Cloud interface. When the trash box is full, the user will get an alert/notification or email on the app interface. They can also manually trigger the lid opening from the app page.
