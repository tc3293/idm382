                #include "thingProperties.h"
                
                // Pin definitions for HC-SR04
                const int trigPin = 5;  // Trigger pin for distance sensor
                const int echoPin = 6;  // Echo pin for distance sensor
                
                // Variable to store distance
                float distance;
                
                // Initialize Arduino IoT Cloud with Wi-Fi
                void setup() {
                  // Start Serial Monitor
                  Serial.begin(9600);
                   // This delay gives the chance to wait for a ..
                   // Serial Monitor without blocking if none is found
                  delay(1500); 
                
                  // Initialize IoT Cloud properties
                  initProperties();
                
                  // Connect to Wi-Fi
                  ArduinoCloud.begin(ArduinoIoTPreferredConnection);
                  
                  // Wait until connected to Arduino IoT Cloud
                  while (ArduinoCloud.connected() != 1) {
                    delay(500);
                  }
                
                  Serial.println("Connected to Arduino IoT Cloud!");
                
                  // Set up the distance sensor pins
                  pinMode(trigPin, OUTPUT);
                  pinMode(echoPin, INPUT);
                }
                
                void loop() {
                  // Update the IoT Cloud
                  ArduinoCloud.update();
                
                  // Read distance from sensor
                  distance = readDistanceSensor();
                
                  // Print distance to Serial Monitor
                  Serial.print("Distance: ");
                  Serial.print(distance);
                  Serial.println(" cm");
                
                  delay(1000);  // Delay to allow easier reading
                }
                
                // Function to calculate distance
                float readDistanceSensor() {
                  // Clear the trigger pin
                  digitalWrite(trigPin, LOW);
                  delayMicroseconds(2);
                
                  // Set the trigger pin HIGH for 10 microseconds
                  digitalWrite(trigPin, HIGH);
                  delayMicroseconds(10);
                  digitalWrite(trigPin, LOW);
                
                  // Read the duration of the echo pulse
                  long duration = pulseIn(echoPin, HIGH);
                
                  // Calculate the distance in cm
                  float distance = duration * 0.034 / 2;
                
                  return distance;
                }

What is it? title?

Arduino IoT cloud, Offline code on the serial monitor

What will it do?


Instead of opening its lid automatically, the garbage can will be revamped. The trash box will detect or monitor the system trash box remotely via an IOT cloud or app, displaying the bin height level and current condition (empty, half-full, full).

How will the end user control it?

The end user can monitor the trash level through an Arduino IoT Cloud interface. When the trash box is full, the user will get an alert/notification or email on the app interface. They can also manually trigger the lid opening from the app page.
